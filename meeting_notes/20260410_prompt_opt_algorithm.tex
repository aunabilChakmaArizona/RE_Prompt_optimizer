\paragraph{Core Idea.} We adopt a two-stage prompt optimization strategy that combines textual and numeric gradients to leverage their complementary strengths. 
Textual gradients enable large improvements in the early stages by making high-level semantic revisions to the prompt; 
however, as optimization approaches a near-optimal region, these updates tend to become unstable and may fluctuate. 
In contrast, numeric gradient-based optimization progresses more slowly but provides fine-grained guidance by identifying which specific tokens are most important to update. 
Therefore, we first apply textual gradient updates to quickly reach a strong initial solution, 
and then refine it using numeric gradients to achieve more stable and precise improvements near the optimum.

*Justification of Line 19 in Algorithm~\ref{alg:numeric-gradient}: \textit{Why we prepend the original prompt as a prefix during autoregression}: 
We append the original prompt prefix along with a paraphrasing instruction to preserve task context during autoregressive generation. 
Using only the prefix of the prompt provides insufficient information about the task and 
ignores the right-side context of the prompt, leading the LLM to generate unrelated or inconsistent tokens. 
Including this context ensures that candidate tokens remain aligned with both the task and the overall prompt structure.

\begin{algorithm}
\caption{Prompt Optimization}
\begin{algorithmic}[1]
\Require Initial prompt $P$, training set $\mathcal{D}_{\text{train}}$, validation set $\mathcal{D}_{\text{val}}$
\Ensure Optimized prompt $P^{*}$

\State $\mathcal{P}_{\text{text}} \gets \{P\}$ \Comment{Textual population}

\For{$t = 1$ to $T$}
    \State Sample $P_t \sim \mathcal{P}_{\text{text}}$
    \State $P' \gets$ TextualGradient$(P_t, \mathcal{D}_{\text{train}})$
    \State Add $P'$ to $\mathcal{P}_{\text{text}}$ with validation score on $\mathcal{D}_{\text{val}}$
\EndFor

\State $P^{*} \gets \arg\max_{P \in \mathcal{P}_{\text{text}}} \text{Score}(P, \mathcal{D}_{\text{val}})$

\State $\mathcal{P}_{\text{num}} \gets \{P^{*}\}$ \Comment{Numeric population}

\For{$q = 1$ to $Q$}
    \State Sample $P_q \sim \mathcal{P}_{\text{num}}$
    \State $\mathcal{P}' \gets$ NumericGradient$(P_q, \mathcal{D}_{\text{train}})$
    \ForAll{$p \in \mathcal{P}'$}
        \State Add $p$ to $\mathcal{P}_{\text{num}}$ with validation score on $\mathcal{D}_{\text{val}}$
    \EndFor
\EndFor

\State $P^{*} \gets \arg\max_{P \in \mathcal{P}_{\text{num}}} \text{Score}(P, \mathcal{D}_{\text{val}})$
\State \Return $P^{*}$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Textual Gradient Update}
\label{alg:textual-gradient}
\begin{algorithmic}[1]
\Require Prompt $P_x$, training set $\mathcal{D}_{\mathrm{train}}$
\Ensure Updated prompt $P_x'$

\State Sample a mini-batch $\mathcal{B} \subset \mathcal{D}_{\mathrm{train}}$
\State Use an LLM to infer predictions for $\mathcal{B}$ under $P_x$
\State Construct a feedback set $\mathcal{D}_{\mathrm{feedback}}$ with three correct and three incorrect predictions
\State Use an LLM to analyze $\mathcal{D}_{\mathrm{feedback}}$ and produce feedback $R$
\State Revise $P_x$ using $R$ to obtain $P_x'$
\State \Return $P_x'$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Numeric Gradient Update}
\label{alg:numeric-gradient}
\begin{algorithmic}[1]
\Require Prompt $P_x$, training set $\mathcal{D}_{\mathrm{train}}$, method $m$, feedback sample size $N$, number of prompt to generate $M$, number of prompts to keep $K$, candidate size $E$
\Ensure Top-$K$ updated prompts $\mathcal{P}'$

\State Sample $N$ examples from $\mathcal{D}_{\mathrm{train}}$ to form $\mathcal{D}_{\mathrm{feedback}}$
\State Convert $\mathcal{D}_{\mathrm{feedback}}$ into binary support--query pairs
\State Run inference on $\mathcal{D}_{\mathrm{feedback}}$ using $P_x$ and obtain predictions $R$
\State Partition $R$ into confusion buckets $\{\mathrm{TP}, \mathrm{TN}, \mathrm{FP}, \mathrm{FN}\}$
\State Construct a balanced subset $\mathcal{D}_{\mathrm{grad}}$ by sampling equally from each bucket
\State Compute token-level gradients of $P_x$ on $\mathcal{D}_{\mathrm{grad}}$
\State Select the top three gradient regions and store them in $\mathcal{R}$
\State Initialize an empty candidate map $\mathcal{C}$

\If{$m = \texttt{GENERATE\_CANDIDATES\_BY\_PROMPTING}$}
    \ForAll{$r \in \mathcal{R}$}
        \State $\widetilde{P}$ = updated prompt $P_x$ where region $r$ is marked using a \texttt{<target>} span
        \State $P_{meta}$ = meta-prompt requesting $E$ replacements (i.e. candidates) of region $r$ in $\widetilde{P}$
        \State Generate candidate tokens $C_r$ using the LLM with $P_{meta}$
        \State Store $C_r$ in $\mathcal{C}$ for region $r$
    \EndFor
\ElsIf{$m = \texttt{GENERATE\_CANDIDATES\_BY\_AUTOREGRESSION}$}
    \ForAll{$r \in \mathcal{R}$}
        \State Let $i$ be the index of region $r$ in $P_x$
        \State Construct $\widetilde{P}$ = $P_x$ + paraphrase instruction + $P_x[0:(i - 1)]$
        \State $\text{Candidate tokens}, C_r = \operatorname*{arg\,topK}_{t \in V} {{LLM}}(t \mid \widetilde{P})$
        \State Store $C_r$ in $\mathcal{C}$ for region $r$
    \EndFor
\EndIf

\State Construct a meta-prompt requesting $M$ combinations of candidates from $\mathcal{C}$
\State Generate $M$ candidate combinations $\mathcal{C}_{\mathrm{comb}}$ by running the meta-prompt
\State Initialize an empty prompt set $\mathcal{P}_{\mathrm{cand}}$

\ForAll{$c \in \mathcal{C}_{\mathrm{comb}}$}
    \State Construct a meta-prompt that replaces each region $r$ in $P_x$ with its selected candidate $c_r$
    \State Generate a new prompt $P_{\mathrm{new}}$ running the meta-prompt
    \State Add $P_{\mathrm{new}}$ to $\mathcal{P}_{\mathrm{cand}}$
\EndFor

\State Initialize an empty score map $\mathcal{S}$
\ForAll{$P \in \mathcal{P}_{\mathrm{cand}}$}
    \State Compute $s = \mathrm{Loss}_{\mathrm{cross\_entropy}}(P, \mathcal{D}_{\mathrm{grad}}) + \lambda \,\mathrm{Perplexity}(P)$
    \State Store score $s$ in $\mathcal{S}$ for prompt $P$
\EndFor

\State Rank prompts in $\mathcal{P}_{\mathrm{cand}}$ according to $\mathcal{S}$
\State Let $\mathcal{P}'$ be the top-$K$ prompts of $\mathcal{P}_{\mathrm{cand}}$
\State \Return $\mathcal{P}'$
\end{algorithmic}
\end{algorithm}
